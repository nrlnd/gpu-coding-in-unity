  currentPosX = (Random.value - 0.5f) * width;

Update 

{ 
  currentPosX += currentSpeed * Time.deltaTime;
  if(currentPosX > width/2) {
  	currentPosX = -width/2;
  }
  
  transform.localPosition = new Vector3(currentPosX, 0.0f, currentPosZ);
}
